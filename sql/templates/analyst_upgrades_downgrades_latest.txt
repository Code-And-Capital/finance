WITH latest_dates AS (
    SELECT
        [TICKER],
        MAX([DATE]) AS max_date
    FROM [dbo].[analyst_upgrades_downgrades]
    WHERE 1=1
    {ticker_filter}
    {date_filter}
    GROUP BY [TICKER]
)
SELECT aud.*
FROM [dbo].[analyst_upgrades_downgrades] AS aud
INNER JOIN latest_dates AS ld
    ON aud.[TICKER] = ld.[TICKER]
    AND aud.[DATE] = ld.max_date
