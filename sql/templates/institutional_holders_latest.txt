WITH latest_dates AS (
    SELECT
        [TICKER],
        MAX([DATE]) AS max_date
    FROM [dbo].[institutional_holders]
    WHERE 1=1
    {ticker_filter}
    {date_filter}
    GROUP BY [TICKER]
)
SELECT ih.*
FROM [dbo].[institutional_holders] AS ih
INNER JOIN latest_dates AS ld
    ON ih.[TICKER] = ld.[TICKER]
    AND ih.[DATE] = ld.max_date
