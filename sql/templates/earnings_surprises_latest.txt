WITH latest_dates AS (
    SELECT
        [TICKER],
        MAX([DATE]) AS max_date
    FROM [dbo].[earnings_surprises]
    WHERE 1=1
    {security_filter}
    {date_filter}
    GROUP BY [TICKER]
)
SELECT es.*
FROM [dbo].[earnings_surprises] AS es
INNER JOIN latest_dates AS ld
    ON es.[TICKER] = ld.[TICKER]
    AND es.[DATE] = ld.max_date
